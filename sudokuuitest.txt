// ito maggegenerate sa 81 na blocks ng sudoku pero wla pang numbers
const grid = document.getElementById('grid');
for (let row = 0; row < 9; row++) {
    for (let col = 0; col < 9; col++) {
        const cell = document.createElement('div');
        cell.classList.add('sudoku-cell');

        const blockRow = Math.floor(row / 3);
        const blockCol = Math.floor(col / 3);
        cell.classList.add((blockRow + blockCol) % 2 === 0 ? 'light' : 'dark');

        if ((col + 1) % 3 === 0 && col !== 8) cell.classList.add('block-right');
        if ((row + 1) % 3 === 0 && row !== 8) cell.classList.add('block-bottom');

        grid.appendChild(cell);
    }
} //kinailangan ko pa to hanapin at mano mano gawin kasi ung container lang nagaappear at di ko alam kung paano gumawa ng 81 blocks HAHSHSAH
